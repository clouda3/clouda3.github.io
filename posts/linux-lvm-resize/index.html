分区不够合理 root 下面空间不够用

umount /home

lvreduce -L 5T /dev/mapper/centos-home

lvextend -L +5T /dev/mapper/centos-root

xfs_growfs /dev/mapper/centos-root

mkfs.xfs -f /dev/mapper/centos-home

mount /home
